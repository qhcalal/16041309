class Person
{
    private int age; //这是一个字段
    public int Age  //这是一个public属性，Age本身不存储任何值
    {
        set //赋值
        {
            if (value < 0)
            {
                return;    //通过属性控制非法值
            }
            age = value; //value表示用户赋过来的值，age不能换成Age，会陷入死循环
        }
        get //取值
        {
            return age;  
        }
      }
   }
